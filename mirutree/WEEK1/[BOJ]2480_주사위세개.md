```js
const fs = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split(" ")
  .map(Number);

const result = (a, b, c) => {
  let rewardType;
  if (a === b && a === c && b === c) {
    rewardType = 1;
    return [a, rewardType];
  } else if (a === b || a === c) {
    rewardType = 2;
    return [a, rewardType];
  } else if (b === c) {
    rewardType = 2;
    return [b, rewardType];
  } else {
    rewardType = 3;
    return [Number(highestNum(a, b, c)), rewardType];
  }
};

const highestNum = (a, b, c) => {
  if (a > b && a > c) {
    return a;
  } else if (b > c) {
    return b;
  } else {
    return c;
  }
};

let rewardCal = result(fs[0], fs[1], fs[2]);
let reward;

if (rewardCal[1] === 1) {
  reward = 10000 + rewardCal[0] * 1000;
} else if (rewardCal[1] === 2) {
  reward = 1000 + rewardCal[0] * 100;
} else {
  reward = rewardCal[0] * 100;
}

console.log(reward);
```
